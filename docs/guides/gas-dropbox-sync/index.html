Dropbox 同期システム 完全復旧マニュアル

Google Drive のファイルを自動的に Dropbox へコピーする仕組みを、 新しいセッションでも最短で再構築するための手順書です。
コードの知識がなくても、順番どおりに進めれば完成します。

⚡ 差分スキップで高速同期 🔒 原本を絶対に壊さない 📦 50MB 超の大容量対応 🗂 階層フォルダも自動追従

📖 このマニュアルについて

このシステムは Google Apps Script（GAS） という、Google が提供する無料の自動化ツールを使っています。難しいサーバーの準備は一切不要で、ブラウザだけで完結します。

以前に設定した内容が消えてしまっても、このマニュアルの手順を上から順に実行すれば、30 分以内に元通りになります。

専門用語はすべて下の「用語集」で解説しています。わからない言葉が出てきたら、まずそちらを確認してください。

📚 よく出てくる言葉の意味

GAS
Google Apps Script の略。Google が無料で提供するプログラム実行環境。ブラウザだけで使え、インストール不要。
App Key / App Secret
Dropbox に「このプログラムは正規のアプリです」と証明するための ID とパスワードのペア。Dropbox の開発者コンソールから取得できる。
認可コード
「このプログラムが自分の Dropbox にアクセスすることを許可する」という一時的な許可証。有効期限があり、一度しか使えない。
リフレッシュトークン
認可コードを使って取得する長期間有効なアクセス許可証。これを保存しておけば、次回以降はブラウザで許可ボタンを押す操作が不要になる。
スクリプト プロパティ
GAS が安全に保持できる「設定値の保管庫」。パスワードのような秘密の情報をコードに直接書かずに済む。
差分スキップ
「すでに Dropbox にある、変わっていないファイル」は転送をスキップする機能。毎回全部送り直すより速く、通信量も少ない。
チャンク送信
大きなファイルを小さなブロックに分けて少しずつ送る方法。GAS には 1 回の処理に制限があるため、これで大容量ファイルも安全に転送できる。
makeCopy()
GAS の命令で「コピーを作る」という意味。原本を直接加工せずコピーだけを操作することで、元のファイルを絶対に壊さない設計になっている。

💻 最終統合版ソースコード

以下のコード全体をコピーして、GAS エディタに貼り付けてください。
「階層同期」「差分スキップ」「50MB 超の大容量チャンク転送」「Google 形式の非破壊分割」の 4 機能がすべて含まれています。

コード.gs — Google Apps Script 
/**
 * Google Drive to Dropbox Sync - Ultimate Recovery Edition
 * 【主要機能】
 * 1. 非破壊分割：原本を makeCopy() してから分割。原本は一切毀損しません。
 * 2. 差分検知：パーツ単位（part1, part2...）で存在チェックを行い、無駄な転送を回避。
 * 3. 大容量対応：通常ファイルはチャンク送信（分割アップロード）でGASの制限を回避。
 */

const CHUNK_SIZE          = 10 * 1024 * 1024; // 10MB 単位
const SIMPLE_UPLOAD_LIMIT = 20 * 1024 * 1024; // 20MB

function syncDriveToDropbox() {
  const props  = PropertiesService.getScriptProperties();
  const config = {
    appKey:       props.getProperty('DROPBOX_APP_KEY'),
    appSecret:    props.getProperty('DROPBOX_APP_SECRET'),
    refreshToken: props.getProperty('DROPBOX_REFRESH_TOKEN'),
    rootFolderId: props.getProperty('DRIVE_FOLDER_ID')
  };

  if (!config.refreshToken)
    return console.error("Error: Refresh token missing. Run setupRefreshToken first.");

  const accessToken = getAccessToken(config.appKey, config.appSecret, config.refreshToken);
  const rootFolder  = DriveApp.getFolderById(config.rootFolderId);
  let   stats        = { success: 0, skipped: 0, error: 0 };

  console.log(`--- [同期開始] ${rootFolder.getName()} ---`);
  processFolder(rootFolder, "", accessToken, stats);
  console.log(`--- [同期終了] 成功:${stats.success} スキップ:${stats.skipped} エラー:${stats.error} ---`);
}

function processFolder(driveFolder, currentPath, accessToken, stats) {
  const files = driveFolder.getFiles();
  while (files.hasNext()) {
    handleFileSync(files.next(), currentPath, accessToken, stats);
  }
  const subFolders = driveFolder.getFolders();
  while (subFolders.hasNext()) {
    const subFolder = subFolders.next();
    const subPath   = currentPath + "/" + subFolder.getName();
    createDropboxFolder(subPath, accessToken);
    processFolder(subFolder, subPath, accessToken, stats);
  }
}

function handleFileSync(file, folderPath, accessToken, stats) {
  const mimeType    = file.getMimeType();
  const isGoogleType = mimeType.includes('google-apps');
  const fileName    = file.getName();

  if (!isGoogleType) {
    const dropboxPath = folderPath + "/" + fileName;
    const meta        = getDropboxMetadata(dropboxPath, accessToken);
    if (meta && meta.size === file.getSize()) { stats.skipped++; return; }
    uploadFile(file.getBlob(), dropboxPath, accessToken, stats);
    return;
  }

  syncGoogleFileWithSplitting(file, folderPath, accessToken, stats);
}

function syncGoogleFileWithSplitting(originalFile, folderPath, accessToken, stats) {
  const mimeType   = originalFile.getMimeType();
  const baseName   = originalFile.getName();
  const targetExt  = (mimeType === MimeType.GOOGLE_DOCS)   ? '.docx'
                    : (mimeType === MimeType.GOOGLE_SHEETS) ? '.xlsx'
                    :                                          '.pptx';
  const targetMime = getTargetMime(mimeType);

  try {
    const dropboxPath = folderPath + "/" + baseName + targetExt;
    if (getDropboxMetadata(dropboxPath, accessToken)) { stats.skipped++; return; }
    uploadFile(exportGoogleFile(originalFile.getId(), targetMime), dropboxPath, accessToken, stats);
  } catch (e) {
    if (e.message.includes("too large")) {
      console.warn(`[制限超過] 分割プロトコル開始: ${baseName}`);
      executeSplitAndUpload(originalFile, 2, targetExt, targetMime, folderPath, accessToken, stats);
    } else {
      console.error(`[エラー] ${baseName}: ${e.message}`);
      stats.error++;
    }
  }
}

function executeSplitAndUpload(originalFile, numParts, ext, targetMime, folderPath, accessToken, stats) {
  const mimeType = originalFile.getMimeType();
  for (let i = 1; i <= numParts; i++) {
    const partName    = `${originalFile.getName()}_part${i}_of_${numParts}${ext}`;
    const dropboxPath = folderPath + "/" + partName;
    if (getDropboxMetadata(dropboxPath, accessToken)) { stats.skipped++; continue; }

    const tempCopy = originalFile.makeCopy(`TEMP_${partName}`);
    try {
      if      (mimeType === MimeType.GOOGLE_DOCS)   truncateDoc(tempCopy.getId(), i, numParts);
      else if (mimeType === MimeType.GOOGLE_SHEETS) truncateSheet(tempCopy.getId(), i, numParts);
      uploadFile(exportGoogleFile(tempCopy.getId(), targetMime), dropboxPath, accessToken, stats);
    } finally {
      tempCopy.setTrashed(true);
    }
  }
}

function truncateDoc(copyId, partIndex, totalParts) {
  const doc         = DocumentApp.openById(copyId);
  const body        = doc.getBody();
  const numChildren = body.getNumChildren();
  const partSize    = Math.floor(numChildren / totalParts);
  const startIdx    = (partIndex - 1) * partSize;
  const endIdx      = (partIndex === totalParts) ? numChildren : partIndex * partSize;
  for (let i = numChildren - 1; i >= 0; i--) {
    if (i < startIdx || i >= endIdx) body.getChild(i).removeFromParent();
  }
  doc.saveAndClose();
}

function truncateSheet(copyId, partIndex, totalParts) {
  const ss = SpreadsheetApp.openById(copyId);
  ss.getSheets().forEach(sheet => {
    const lastRow  = sheet.getLastRow();
    if (lastRow <= 1) return;
    const partSize = Math.floor(lastRow / totalParts);
    const startRow = (partIndex - 1) * partSize + 1;
    const endRow   = (partIndex === totalParts) ? lastRow : partIndex * partSize;
    if (endRow < lastRow)  sheet.deleteRows(endRow + 1, lastRow - endRow);
    if (startRow > 1)     sheet.deleteRows(1, startRow - 1);
  });
  SpreadsheetApp.flush();
}

function uploadFile(blob, path, accessToken, stats) {
  const bytes = blob.getBytes();
  if (bytes.length <= SIMPLE_UPLOAD_LIMIT) uploadSimple(bytes, path, accessToken);
  else                                       uploadChunked(bytes, path, accessToken);
  stats.success++;
}

function uploadChunked(bytes, path, accessToken) {
  const totalSize = bytes.length;
  const startRes  = UrlFetchApp.fetch('https://content.dropboxapi.com/2/files/upload_session/start', {
    method:  'post',
    headers: {
      'Authorization':   'Bearer ' + accessToken,
      'Dropbox-API-Arg': JSON.stringify({ close: false }),
      'Content-Type':    'application/octet-stream'
    },
    payload: bytes.slice(0, CHUNK_SIZE)
  });
  const sessionId = JSON.parse(startRes.getContentText()).session_id;
  let   offset    = CHUNK_SIZE;

  while (offset + CHUNK_SIZE < totalSize) {
    UrlFetchApp.fetch('https://content.dropboxapi.com/2/files/upload_session/append_v2', {
      method:  'post',
      headers: {
        'Authorization':   'Bearer ' + accessToken,
        'Dropbox-API-Arg': JSON.stringify({ cursor: { session_id: sessionId, offset }, close: false }),
        'Content-Type':    'application/octet-stream'
      },
      payload: bytes.slice(offset, offset + CHUNK_SIZE)
    });
    offset += CHUNK_SIZE;
  }

  UrlFetchApp.fetch('https://content.dropboxapi.com/2/files/upload_session/finish', {
    method:  'post',
    headers: {
      'Authorization':   'Bearer ' + accessToken,
      'Dropbox-API-Arg': JSON.stringify({
        cursor: { session_id: sessionId, offset },
        commit: { path, mode: 'overwrite', mute: true }
      }),
      'Content-Type':    'application/octet-stream'
    },
    payload: bytes.slice(offset)
  });
}

function uploadSimple(bytes, path, accessToken) {
  UrlFetchApp.fetch('https://content.dropboxapi.com/2/files/upload', {
    method:  'post',
    headers: {
      'Authorization':   'Bearer ' + accessToken,
      'Dropbox-API-Arg': JSON.stringify({ path, mode: 'overwrite', mute: true }),
      'Content-Type':    'application/octet-stream'
    },
    payload: bytes
  });
}

function getDropboxMetadata(path, accessToken) {
  const res = UrlFetchApp.fetch('https://api.dropboxapi.com/2/files/get_metadata', {
    method:             'post',
    headers:            { 'Authorization': 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    payload:            JSON.stringify({ path }),
    muteHttpExceptions: true
  });
  return res.getResponseCode() === 200 ? JSON.parse(res.getContentText()) : null;
}

function createDropboxFolder(path, accessToken) {
  UrlFetchApp.fetch('https://api.dropboxapi.com/2/files/create_folder_v2', {
    method:             'post',
    headers:            { 'Authorization': 'Bearer ' + accessToken, 'Content-Type': 'application/json' },
    payload:            JSON.stringify({ path }),
    muteHttpExceptions: true
  });
}

function exportGoogleFile(fileId, targetMime) {
  const url = `https://www.googleapis.com/drive/v3/files/${fileId}/export?mimeType=${targetMime}`;
  const res = UrlFetchApp.fetch(url, {
    headers:            { Authorization: 'Bearer ' + ScriptApp.getOAuthToken() },
    muteHttpExceptions: true
  });
  if (res.getResponseCode() !== 200) throw new Error("Google Export Error: " + res.getResponseCode());
  return res.getBlob();
}

function getTargetMime(mime) {
  if (mime === MimeType.GOOGLE_DOCS)   return 'application/vnd.openxmlformats-officedocument.wordprocessingml.document';
  if (mime === MimeType.GOOGLE_SHEETS) return 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet';
  if (mime === MimeType.GOOGLE_SLIDES) return 'application/vnd.openxmlformats-officedocument.presentationml.presentation';
  return null;
}

function getAccessToken(key, secret, refresh) {
  const res = UrlFetchApp.fetch('https://api.dropboxapi.com/oauth2/token', {
    method:  'post',
    payload: {
      grant_type:    'refresh_token',
      refresh_token: refresh,
      client_id:     key,
      client_secret: secret
    }
  });
  return JSON.parse(res.getContentText()).access_token;
}

function setupRefreshToken() {
  const props   = PropertiesService.getScriptProperties();
  const payload = {
    code:          props.getProperty('DROPBOX_AUTH_CODE'),
    grant_type:    'authorization_code',
    client_id:     props.getProperty('DROPBOX_APP_KEY'),
    client_secret: props.getProperty('DROPBOX_APP_SECRET')
  };
  const res  = UrlFetchApp.fetch('https://api.dropboxapi.com/oauth2/token', {
    method: 'post', payload, muteHttpExceptions: true
  });
  const json = JSON.parse(res.getContentText());
  if (json.refresh_token) {
    props.setProperty('DROPBOX_REFRESH_TOKEN', json.refresh_token);
    console.log("【成功】リフレッシュトークンを保存しました。");
  } else {
    console.error("【失敗】トークン取得エラー。認可コードが古い可能性があります。");
  }
}
